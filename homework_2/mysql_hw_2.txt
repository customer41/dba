1. Создание таблицы для хранения товаров

CREATE TABLE `products` (
    `id` SERIAL,
    `title` VARCHAR(255),
    `article` VARCHAR(20) NOT NULL,
    `img` VARCHAR(1024),
    `price` INT UNSIGNED,
    `start_date` DATE,
    `quantity` INT UNSIGNED
    );

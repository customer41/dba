1. Создание таблиц

CREATE TABLE `books` (
    `id` SERIAL,
    `title` VARCHAR(100),
    `year` INT,
    `author` VARCHAR(100),
    `price` INT
);

CREATE TABLE `publishers` (
    `id` SERIAL,
	`title` VARCHAR(100),
	`establishment_year` INT
);
